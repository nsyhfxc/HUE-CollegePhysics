\documentclass{xStandalone}

\newcommand{\lne}[1]
{
    \draw[-,ultra thin,red] (#1,-0.1)--(#1,0.1);
}

\newcommand{\pnt}[1]
{
    \draw[-,ultra thin,red,fill=red!70!white] (#1,0) circle (0.04);
}

\begin{document}
\begin{tikzpicture}

\tikzset{axi/.style={gray,ultra thin}}

\draw[axi] (-1.5,-1.5) rectangle (8.3,1.5);

\path (0,0) coordinate (O);

\draw[axi,->] ($(O)+(-1.3,0)$)--($(O)+(+1.3,0)$) node[above] {\tiny $x$};
\draw[axi,->] ($(O)+(0,-1.3)$)--($(O)+(0,+1.3)$) node[right] {\tiny $y$};
    
\draw[fill=black,fill opacity=0.1] (O) circle (1);

\draw[thick,latex-latex] ($(O)+(0,-0.4)$)--($(O)+(0,+0.4)$);
\draw[thick,latex-latex] ($(O)+(-0.7,0)$)--($(O)+(+0.7,0)$);

\path (O) node[point]{};

\draw[axi,->] ($(O)+(2,0)$) coordinate (A) -- ++(6,0) coordinate (B) node[above] {\tiny $z$};

\begin{scope}[decoration={markings,
    mark=
    between positions 0.125 and 0.9 step 0.8cm
    with
    {
        \lne{-0.3}
        \pnt{-0.1}
        \pnt{+0.1}
        \pnt{+0.3}
    }}]
    \draw[-latex,thick,postaction={decorate}] 
    ($(A)+(0.2,0)$) coordinate (A1)--
    ($(B)+(-0.2,0)$) coordinate (B1);
\end{scope}

\path ($(A1)!0.5!(B1)$) node[below=0.5cm] {\footnotesize 光线传播方向};

\end{tikzpicture}    
\end{document}
