\NewDocumentCommand{\frameb}{}
{
    \draw[dotted] (-2.0,-0.1) rectangle (2.0,2.6);
    \draw (-1.5,0) rectangle (1.5,2.5);
}

\NewDocumentCommand{\ele}{mm}
{
    \lPoint{O}{#1}
    \lShift{P}{O}{#2:0.3}
    \lShift{N}{O}{#2:-0.3}
    \lShift{Pt}{O}{#2:0.25}
    \lShift{Nt}{O}{#2:-0.25}
    \node at (P) {\tiny $+$};
    \node at (N) {\tiny $-$};
    \draw (P) circle (0.05);
    \draw (N) circle (0.05);
    \draw[thin,-latex] (Nt)--(Pt);
}
