\NewDocumentCommand{\GraphLSR}{mmmm}
{
    \makecell*
    {%
        \begin{tikzpicture}[scale=0.45]%
            \draw[very thin,gray](-1.2,0)--(1.2,0);%
            \draw[very thin,gray](0,-1.2)--(0,1.2);%
            \draw[very thin] (1.2,1.2) rectangle (-1.2,-1.2);%
            \draw[very thin,gray,dashed] (1.0,1.0) rectangle (-1.0,-1.0);%
            \draw[blue!50!white,domain=0:pi,->,samples=200] plot ({cos((#1*\x r)+#3)},{cos((#2*\x r)+#4)});%
            \draw[blue!50!white,domain=pi:2*pi,-,samples=200] plot ({cos((#1*\x r)+#3)},{cos((#2*\x r)+#4)});%
            \draw[very thin,domain=0:2*pi,fill=white] plot ({(0.03*cos(\x r)+cos(#3))},{0.03*sin(\x r)+cos(#4)});%
        \end{tikzpicture}%
    }%
}
