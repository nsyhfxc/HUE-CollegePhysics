\NewDocumentCommand{\DrawDouble}{O{0}}
{
    \lPoint{O}{0,0}
    \lShift{A}{O}{#1:3.5}
    \lShift{B}{O}{#1:-3.5}
    \lShift{A1}{A}{\fpeval{#1+90}:0.25}
    \lShift{A2}{A}{\fpeval{#1-90}:0.25}
    \lShift{B1}{B}{\fpeval{#1+90}:0.25}
    \lShift{B2}{B}{\fpeval{#1-90}:0.25}
    \draw[thin] (A1)--(B1);
    \draw[thin] (A2)--(B2);

    \lShift{Fa}{A}{0,1}
    \lShift{Fb}{B}{0,-1}
    \draw[-latex] (A)--(Fa) node[right] {$\vb*{F}$};
    \draw[-latex] (B)--(Fb) node[right] {$\vb*{F}$};

    \lShift{A3}{A}{\fpeval{#1+90}:0.4}
    \lShift{A4}{A}{\fpeval{#1-90}:0.4}
    \lShift{B3}{B}{\fpeval{#1+90}:0.4}
    \lShift{B4}{B}{\fpeval{#1-90}:0.4}
    \draw[gray,ultra thin,-latex] (A3)--(B3);
    \draw[gray,ultra thin,latex-,dashed] (A4)--(B4);

    \draw[fill=white] (A) circle (0.25) node {\large $\cdot$};
    \draw[fill=white] (B) circle (0.25) node {\large $\times$};

    \lShift{d}{O}{#1:3.25}
    \draw[-latex,cyan] (O)--(d) node [above left] {$\frac{\vb*{d}}{2}$};

    \lShift{D}{O}{\fpeval{#1-90}:2}
    \draw[-latex,thick] (O)--(D) node[below right] {$\vb*{m}$};
    \draw[fill=white] (O) circle (0.05);
}
