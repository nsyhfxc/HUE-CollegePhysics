\NewDocumentCommand{\frameb}{}
{
    \foreach \x in {0.05,0.25,...,2.45}
    {
        \draw[gray,thin] (-2,\x)--(-1.5,\x);
        \draw[-latex,gray,thin] (1.5,\x)--(2,\x);
    }
    \draw[dotted] (-2.0,-0.1) rectangle (2.0,2.6);
    \draw[fill=white] (-1.5,0) rectangle (1.5,2.5);
    \foreach \x in {0.15,0.35,...,2.36}
    {
        \node[right] at (-1.52,\x) {\scriptsize $-$};
        \node[left]  at (+1.52,\x) {\scriptsize $+$};
    }
}

\NewDocumentCommand{\ele}{mm}
{
    \lPoint{O}{#1}
    \lShift{P}{O}{#2:0.3}
    \lShift{N}{O}{#2:-0.3}
    \lShift{Pt}{O}{#2:0.25}
    \lShift{Nt}{O}{#2:-0.25}
    \node at (P) {\tiny $+$};
    \node at (N) {\tiny $-$};
    \draw (P) circle (0.05);
    \draw (N) circle (0.05);
    \draw[thin,-latex] (Nt)--(Pt);
}
