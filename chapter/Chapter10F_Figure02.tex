\NewDocumentCommand{\DrawDouble}{O{0}}
{
    \lPoint{O}{0,0}
    \lShift{A}{O}{#1:3.5}
    \lShift{B}{O}{#1:-3.5}
    \lShift{A1}{A}{\fpeval{#1+90}:0.25}
    \lShift{A2}{A}{\fpeval{#1-90}:0.25}
    \lShift{B1}{B}{\fpeval{#1+90}:0.25}
    \lShift{B2}{B}{\fpeval{#1-90}:0.25}
    \draw[thin] (A1)--(B1);
    \draw[thin] (A2)--(B2);

    \draw[fill=white] (A) circle (0.25) node {\large $\cdot$};
    \draw[fill=white] (B) circle (0.25) node {\large $\times$};

    \lShift{D}{O}{\fpeval{#1-90}:2}
    \draw[-latex,thick] (O)--(D) node[below right] {$\vb*{m}$};
    \draw[fill=white] (O) circle (0.05);
}

\NewDocumentCommand{\DrawDoubleI}{O{0}}
{
    \lPoint{O}{0,0}
    \lShift{A}{O}{#1:3.5}
    \lShift{B}{O}{#1:-3.5}
    \lShift{A1}{A}{\fpeval{#1+90}:0.25}
    \lShift{A2}{A}{\fpeval{#1-90}:0.25}
    \lShift{B1}{B}{\fpeval{#1+90}:0.25}
    \lShift{B2}{B}{\fpeval{#1-90}:0.25}
    \draw[very thin,dashed,black!20!white] (A1)--(B1);
    \draw[very thin,dashed,black!20!white] (A2)--(B2);

    \draw[black!20!white,fill=white,very thin] (A) circle (0.25) node {\large $\cdot$};
    \draw[black!20!white,fill=white,very thin] (B) circle (0.25) node {\large $\times$};

    \lShift{Fa}{A}{0,1}
    \lShift{Fb}{B}{0,-1}
    \lShift{Fa1}{A}{#1:0.866}
    \lShift{Fb1}{B}{#1:-0.866}
    \lShift{Fa2}{A}{\fpeval{#1+90}:0.5}
    \lShift{Fb2}{B}{\fpeval{#1+90}:-0.5}
    \draw[-latex,black!20!white,ultra thin] (A)--(Fa);
    \draw[-latex,black!20!white,ultra thin] (B)--(Fb);
    \draw[-latex,black!20!white,ultra thin] (A)--(Fa1);
    \draw[-latex,black!20!white,ultra thin] (B)--(Fb1);
    \draw[-latex,cyan,ultra thin] (A)--(Fa2);
    \draw[-latex,cyan,ultra thin] (B)--(Fb2);
    \draw[dotted,gray,very thin] (Fa1)--(Fa)--(Fa2);
    \draw[dotted,gray,very thin] (Fb1)--(Fb)--(Fb2);
    \node[cyan] at (0,0) {\Large $\circlearrowleft$};
}
