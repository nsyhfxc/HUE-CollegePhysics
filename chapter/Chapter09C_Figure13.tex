\NewDocumentCommand{\ele}{mm}
{
    \lPoint{O}{#1}
    \lShift{P}{O}{#2:0.85}
    \lShift{N}{O}{#2:-0.85}
    \lShift{Pt}{O}{#2:0.75}
    \lShift{Nt}{O}{#2:-0.75}
    \node at (P) {\tiny $+$};
    \node at (N) {\tiny $-$};
    \draw (P) circle (0.1);
    \draw (N) circle (0.1);
    \draw[thin,-latex] (Nt)--(Pt);
}
