\NewDocumentCommand{\drawunit}{mm}
{
    \lPoint{O}{#1}
    \lShift{A}{O}{#2:-0.12}
    \draw[rotate=#2,-stealth,thin](A) arc (180:540:0.12 and 0.24);
    \lShift{B}{O}{#2:0.3}
    \draw[-latex,gray,ultra thin] (O)--(B);
}
