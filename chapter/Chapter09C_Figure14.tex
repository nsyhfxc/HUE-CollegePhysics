\NewDocumentCommand{\ele}{mm}
{
    \lPoint{O}{#1}
    \lShift{P}{O}{#2:0.68}
    \lShift{N}{O}{#2:-0.68}
    \lShift{Pt}{O}{#2:0.58}
    \lShift{Nt}{O}{#2:-0.58}
    \node at (P) {\tiny $+$};
    \node at (N) {\tiny $-$};
    \draw (P) circle (0.1);
    \draw (N) circle (0.1);
    \draw[thin,-latex] (Nt)--(Pt);
    \draw[red!50!white,fill=red!80!white,very thin] (O) circle (0.015);
}
